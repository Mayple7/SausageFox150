Definition at line 169 of file StartSlides.c.

{
        // Fade out the slide
        if(fade == 2)
                alpha -= 0.01;
        // Fade in the slide
        else if(fade == 1)
                alpha += 0.01;
        else
                alpha = 1.0;

        // The fade in is done
        if(alpha > 1.0)
        {
                alpha = 1.0f;
                fade = 0;
        }
        // The fade out is done and it is not the last slide
        else if(alpha < 0.0 && slideTextureNum != 2)
        {
                slideTextureNum += 1;
                fade = 1;
                alpha = 0.0;
        }
        // The fade out is done, go to main menu
        else if(alpha < 0.0)
        {
                return 1;
        }

        // Timer for the slide staying visible
        if(fade == 0)
                slideTimer += 1;

        // Start fading out the slide after some time
        if(slideTimer == 120)
        {
                fade = 2;
                slideTimer = 0;
        }

        // Do not change levels
        return 0;
}
